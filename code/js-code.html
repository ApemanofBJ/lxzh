代码块隐藏

		function showcontent(str){
			var sub_menu = document.getElementById(str);
			var dis_v = sub_menu.style.display;

			if (dis_v == "block")
				sub_menu.style.display = "none";
			else 
				sub_menu.style.display = "block";		
		}

		function show_content()  //显示内容
		{
			document.getElementById("id").style.display="block";
			document.getElementById("id").innerHTML="收起-"；
			document.getElementById("id").href="javascript:hide_content()";
		}

		function hide_content() //隐藏内容
		{
			document.getElementById("id").style.display="none";
			document.getElementById("id").innerHTML="更多选项-"；
			document.getElementById("id").href="javascript:show_content()";
		}

数据验证

		<p>请输入 1 与 10 之间的数：</p>

		<input id="numb">

		<button type="button" onclick="myFunction()">提交</button>

		<p id="demo"></p>

		<script>
		function myFunction() {
		  var x, text;

		  // 获取 id="numb" 的输入字段的值
		  x = document.getElementById("numb").value;

		  // 如果 x 不是数字或小于 1 或大于 10
		  if (isNaN(x) || x < 1 || x > 10) {
		    text = "输入无效";
		  } else {
		    text = "输入有效";
		  }
		  document.getElementById("demo").innerHTML = text;
		}
		</script>
