第一章 字符串函数的实现

重点：C语言字符串以0作为结尾

字符串长度函数

		size_t myStrlen(const char *s)		//const char* 代表常量字符串
		{
		    if (s == NULL)
		        return -1;

		    const char *p = s; 				//const char* 可以读取指针但是不能进行修改
		    while (*p++ != '\0')            //先进行取值计算，再进行指针的++计算
		        ;							//空语句

		    return p-s-1;					//数组的指针算术计算
		}

字符串复制函数

		char* myStrcpy(char *dst,const char *src)
		{
		    //函数参数检查
		    char *tmp = dst;

		    while ((*tmp++ = *src++)!='\0')
		        ;

		    return tmp;
		}
